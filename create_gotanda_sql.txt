create table gotanda.member (
	member_id int not null auto_increment,
	name varchar(255),
	address varchar(255),
	tel varchar(255),
	sex varchar(1),
	birthday varchar(8),
	
	primary key (member_id)
)

create table gotanda.menu (
	menu_id int not null auto_increment,
	syohin_name varchar(255),
	tanka int,
	primary key (id)
);


create table gotanda.tenpo (
	tenpo_id int not null auto_increment,
	tenpo_name varchar(255),
	tenpo_address varchar(255),
	tenpo_tel varchar(255),
	primary key (tenpo_id)
);


create table gotanda.kaikei (
	kaikei_id int not null auto_increment,
	member_id varchar(10),
	tenpo_id varchar(10),
	raiten_day varchar(8),
	sum_kingaku int,
	primary key (kaikei_id)
);

create table gotanda.meisai (
	kaikei_id varchar(10),
	menu_id varchar(10)
)
